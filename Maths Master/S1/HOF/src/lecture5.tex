\begin{definition}[Filtered groups and lattices]
\begin{itemize}[topsep=-6pt, itemsep=0pt]
    \item Prefiltration $G_0 \geq G_1 \geq \cdots , \quad  [G_i, G_j] \subseteq G_{i+j}$
    \item Filtration if $G_0 = G_1$
    \item $ \Gamma \leq G$ filtered lattice if discrete and $\Gamma _i \coloneqq \Gamma \cap G_i$ cocompact in $G_i$
\end{itemize}
\end{definition}

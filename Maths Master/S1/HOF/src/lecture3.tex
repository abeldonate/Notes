\begin{theorem}(Inverse 4APs) If
$\lvert \sum_{x, y \in G} f_0(x) f_1(x+y)f_2(x+2y) \rvert \geq \varepsilon p^{2n} \quad  \Rightarrow \quad \ \exists a_i \in \mathbb{F}_p^{n} \text{ st }$, then
\[
\ \exists M_i, b_i \text{ st } \quad \lvert \sum_{x \in G}f_i(x)e_p(x^T M_i x + b_i^T x) \rvert \geq  c_p e^{-C \varepsilon ^{-C}} p^n
\]
\end{theorem}

\begin{theorem}(Inverse Gowers $U^3$) If $\lVert f \rVert_{U^{3}} \geq \varepsilon p^{n/2}$, then
\[
\ \exists M, b \text{ st } \quad \lvert \sum_{x \in \mathbb{F}_p^n} f(x)e_p(x^TMx + b^Tx) \rvert \geq c_p e^{-C\varepsilon ^{-C}}p^n
\]
\end{theorem}

$c_p, C$ absolute depending only on $p$.

\begin{proposition}
$\mathcal{H} \subseteq \mathbb{F}_p^{n}, \lvert \mathcal{H}  \rvert \geq \delta p^{n}$. Suppose that $\ \forall h \in \mathcal{H} \ \exists \varphi (h) $ such that

\[

\]
\end{proposition}

\begin{definition}[Additive energy]
$E(A) = \lvert \{(x_1, \ldots, x_4) \in A^4 : x_1 + x_2 = x_3 + x_4\} \rvert $ 
\end{definition}

$E(A) \leq \lvert A \rvert^{3} $. Large additive energy if $E(A) \geq \frac{\lvert A \rvert ^3}{K}$

\begin{theorem}[Balog-Szemeredi-Gowers]
$G$ discrete abelian group, $A \subseteq G$ finite, $E(A) \geq \frac{\lvert A \rvert ^{3} }{K}$. Then $\ \exists A' \subseteq A$ st
\[
\lvert A' \rvert \gg K^{-O(1)} \lvert A \rvert , \quad  \text{ and } \quad \lvert A' + A' \rvert \ll K^{O(1)} \lvert A' \rvert 
\]
\end{theorem}

\begin{theorem}[Freiman-Ruzsa] If $\lvert A + A \rvert \leq K \lvert A \rvert \quad  \Rightarrow \lvert \langle A \rangle \rvert \geq e^{O_p(K^{O(1)})} \lvert A \rvert $
\end{theorem}

$A = \bigoplus_{d \geq 0} A_d$ graded ring, $A_0$ subring, $A_+ = \bigoplus_{d >0} A_d$ irrelevant ideal.

\begin{definition}[Homo. ideal] 
$f \in I \Rightarrow f_{d_1}, \ldots, f_{d_r} \in I \Leftrightarrow  I = \bigoplus_{d \geq 0} (I \cap A_d) \Leftrightarrow I = (f_1, \ldots, f_r), f_i$ homo.
\end{definition}

\begin{definition}[Homogenization]
$I ^{h} = \bigoplus_{d \geq 0} (I \cap A_d)$
\end{definition}

\begin{proposition}
$I$ homo $ \Rightarrow \sqrt{I}$ homo.
\end{proposition}

\begin{definition}
$M_{(a)} \coloneqq (M_a)_0, \quad  M_{(\mathfrak{p})} \coloneqq (M_{\mathfrak{p}})_0$. e.g. $A = k[x_0, \ldots, x_n], A_{(x_i)} = k[\frac{x_0}{x_i}, \ldots, \frac{x_n}{x_i}]$
\end{definition}

$D_+(f), f \in A_+$ homo, basis of topology.

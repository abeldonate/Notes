$A = \bigoplus_{d \geq 0} A_d$ graded ring, $A_0$ subring.

\begin{proposition} $X$ has a generic point $ \Leftrightarrow $ $X$ irreducible.
\end{proposition}

\begin{definition}[Homo. ideal] 
$f \in I \Rightarrow f_{d_1}, \ldots, f_{d_r} \in I \Leftrightarrow  I = \bigoplus_{d \geq 0} (I \cap A_d) \Leftrightarrow I = (f_1, \ldots, f_r), f_i$ homo.
\end{definition}

\begin{definition}[Homogenization]
$I ^{h} = \bigoplus_{d \geq 0} (I \cap A_d)$
\end{definition}

\begin{definition}[Irrelev. ideal] $I = \bigoplus_{d > 0} A_d$
\end{definition}

\begin{proposition}
$I$ homo $ \Rightarrow \sqrt{I}$ homo.
\end{proposition}

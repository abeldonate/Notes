$A = \bigoplus_{d \geq 0} A_d$ graded ring, $A_0$ subring, $A_+ = \bigoplus_{d >0} A_d$ irrelevant ideal.

\begin{proposition} $X$ has a generic point $ \Leftrightarrow $ $X$ irreducible.
\end{proposition}

\begin{definition}[Homo. ideal] 
$f \in I \Rightarrow f_{d_1}, \ldots, f_{d_r} \in I \Leftrightarrow  I = \bigoplus_{d \geq 0} (I \cap A_d) \Leftrightarrow I = (f_1, \ldots, f_r), f_i$ homo.
\end{definition}

\begin{definition}[Homogenization]
$I ^{h} = \bigoplus_{d \geq 0} (I \cap A_d)$
\end{definition}

\begin{proposition}
$I$ homo $ \Rightarrow \sqrt{I}$ homo.
\end{proposition}

\begin{proposition}
$D_+(f), f \in A_+$ homo, basis of topology.
\end{proposition}

$\boxed{\Proj A \coloneqq \{ \mathfrak{p} \text{ homo}, A_+ \not\subseteq \mathfrak{p} \}} $
$ \boxed{V_+(I) \coloneqq \{ \mathfrak{p} \in \Proj A : I \subseteq \mathfrak{p} \}} $
$\boxed{ D_+(f) \coloneqq \{ \mathfrak{p} \in \Proj A : f \notin \mathfrak{p} \}} $

$\boxed{ V_+(I \cap J) = V_+(I) \cup V_+(J)}$
$\boxed{V_+(\sum I_i) = \bigcap V_+(I_i)}$
$ \boxed{V_+(A_+) = V_+(A) = \emptyset} $
$\boxed{V_+((0)) = \Proj A}$

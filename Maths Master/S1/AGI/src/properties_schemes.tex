We state $(U_i)$ open cover, $(W_i)$ open affine cover and $U$ open subset, $Z$ closed subset.

Also, $x \in X$, $y \in Y$ arbitrary point, $\eta \in X$ the generic point, $f : X \to Y$.

\begin{definition}
\begin{itemize}[topsep=-6pt, itemsep=0pt]
    \item (Disconected) $X = U_1 \cup  U_2$, $U_1 \cap U_2 = \emptyset$. e.g. $\Spec(k\times k)$.
    \item (Reducible) $X = Z_1 \cup Z_2$, $Z_1, Z_2$. e.g. $\Spec(k[x, y]/(xy))$.
    \item (Quasi-compact) $X = \bigcup U_i$, finite subcover. e.g. not $\Spec(k[x_1, x_2, ...]) - \{0\}$.
    \item (Quasi-separated) $U_i, U_j$ quasi-compact $ \Rightarrow $ $U_i \cap U_j$ quasi-compact. e.g. $U_i = \mathbb{A}_k^{ \infty}$ glued along $\mathbb{A}_k^{\infty} - \{0\} \Rightarrow U_1 \cap U_2 = \mathbb{A} - \{0\}$.
\end{itemize}
\end{definition}

\begin{proposition}
irreducible $ \Rightarrow $ connected 
\end{proposition}

\begin{proposition} $X$ has a generic point $ \Leftrightarrow $ $X$ irreducible.
\end{proposition}

\begin{proposition}
Affine $ \Rightarrow $ quasi-compact + quasi-separated
\end{proposition}

\begin{definition}[Open immersion]
$f: (U, \mathcal{O}_X|_U ) \to (X, \mathcal{O}_X), f^\#$ iso
\end{definition}

\begin{definition}[Closed immersion]
$\iota : Z \hookrightarrow X$, $ \begin{cases}
\iota \text{ homeo. onto closed subset} \\
\iota ^\# : \mathcal{O}_Y \twoheadrightarrow \iota_* \mathcal{O}_X
\end{cases}  $
\end{definition}

\begin{proposition}
Closed subschemes of aff. are aff.
\end{proposition}

\begin{proposition}
$D(f) \to X$ open immersion, $V(I) \to X$ closed immersion.
\end{proposition}

\begin{proposition}
$\Spec B \to \Spec A$ closed imm. $ \Leftrightarrow $ $ \varphi : A \to B$ surjective.
\end{proposition}

\begin{proposition}
$Z \subseteq \Spec(A) \Leftrightarrow Z = \Spec (A/I)$.
\end{proposition}

\begin{definition}
\begin{itemize}[topsep=-6pt, itemsep=0pt]
	\item (affine-local) $(a_1, \ldots, a_n) = A$, $A \in \mathcal{P} \Leftrightarrow A_{a_i} \in \mathcal{P}$
	\item (stalk-local) $A \in \mathcal{P} \Leftrightarrow A_{\mathfrak{p}} \in \mathcal{P}$
\end{itemize}
\end{definition}

\begin{proposition}
Stalk -local $ \Rightarrow $ affine-local ($A_{\mathfrak{p}} \cong (A_a)_{\mathfrak{p}A_a}$)
\end{proposition}

\begin{definition}
	\begin{itemize}[topsep=-6pt, itemsep=0pt]
		\item $X$ Noetherian $ \Leftrightarrow $ quasi-compact + $A_i$ Noeth. (affine-local) (e.g. not stalk-local $A=\Pi_{\infty}\mathbb{F}_2$)
		\item $X$ integral $ \Leftrightarrow \mathcal{O}_X(U)$ integral $ \Leftrightarrow X $ connected + $\mathcal{O}_{X,x}$ integral $ \Leftrightarrow X$ reduced + irred.
		\item $X$ reduced $ \Leftrightarrow \mathcal{O}_X(U)$ reduced (stalk-local)
		\item $X$ normal $ \Leftrightarrow A_{\mathfrak{p}}$ integral and integrally closed (stalk-local)
	\end{itemize}
\end{definition}

\begin{proposition} $\Spec (A)$ irred. $ \Leftrightarrow A/\sqrt{(0)}$ integral.
\end{proposition}

\begin{definition}[Residue / Function field]
$k(x) = O_{X, x}/\mathfrak{m}_x$, $k(X) = O_{X, \eta}$
\end{definition}

\begin{proposition} $\mathcal{O}_X (U) = \bigcap_{x \in U} \mathcal{O}_{X,x}$
\end{proposition}

\begin{definition}[Ideal sheaf]
$\mathcal{I} : U \mapsto \mathcal{I}(U) \subseteq \mathcal{O}_X(U)$ ideal.
\end{definition}

\begin{definition}
\begin{itemize}[topsep=-6pt, itemsep=0pt]
	\item (fiber) $X_y \coloneqq f^{-1}(y) = X \times_Y \Spec k(y)$
	\item (geometric fiber) $X_{\overline{y}} \coloneqq X \times_Y \Spec \overline{k(y)}$
	\item (diagonal morphism) $\Delta_{X / Y} \coloneqq X \to X \times_Y X$, $p_1 \circ \Delta = p_2 \circ \Delta = \id_X$
	\item (graph) $\Gamma_f \coloneqq (\id, f) : X \to X \times_Y Y$
\end{itemize}
\end{definition}

\begin{definition} $\mathcal{P}$ satisfies
\begin{itemize}[topsep=-6pt, itemsep=0pt]
	\item (COMP) $f, g \in \mathcal{P} \Rightarrow g \circ f \in \mathcal{P}$
	\item (CANC) $g \circ f \in \mathcal{P}  \Rightarrow f \in \mathcal{P} $ 
	\item (BC) $f \in \mathcal{P} \Rightarrow f_{Y'} \in \mathcal{P}$
	\item (LOCT) $f \in \mathcal{P} \Leftrightarrow f_{V_i} \in \mathcal{P} \ \forall i \ ( \bigcup V_i = Y)$
	\item (LOCS) $f \in \mathcal{P} \Leftrightarrow f_\mid {U_i} \in \mathcal{P} \ \forall i \ ( \bigcup U_i = X)$
\end{itemize}
\end{definition}

\begin{definition} $f \in \Delta_{\mathcal{P} } \Leftrightarrow  \Delta_{X/Y} \in \mathcal{P} $
\end{definition}

\begin{proposition} 
\begin{itemize}[topsep=-6pt, itemsep=0pt]
	\item $\mathcal{P} $ (BC) $ \Rightarrow $ $\Delta_{\mathcal{P}} $ (BC)
	\item $ \mathcal{P} $ (BC, COMP) $ \Rightarrow $ $\Delta_{\mathcal{P}} $ (COMP)
	\item $ \mathcal{P} $ (LOCT) $ \Rightarrow $ $\Delta_{\mathcal{P}} $ (LOCT)
	\item $ \mathcal{P} $ (BC, COMP), $f, g \in \mathcal{P} \Rightarrow  f \in \mathcal{P}$ 
	\item $ \mathcal{P} $ (BC, COMP), $ g \circ f, \Delta _{Y/Z} \in \mathcal{P}   \Rightarrow f \in \mathcal{P} $

\begin{definition}

	\item (quasi-separated) $\Delta_{X/Y}$ quasi-compact
	\item (finite fibers) $ f^{-1}(y)$ finite set
\end{itemize}
\end{definition}

	\item closed

\begin{definition}

	\item (immersion) $f = g \circ h$, $g, h$ open and closed imm. resp.
	\item (loc. fin. type) $V = \Spec B \Rightarrow f ^{-1}(V) = \bigcup \Spec A_i, A_i$ f.g. $B$-alg.

	\item (affine) $V$ affine $ \Rightarrow f^{-1}(V)$ affine
	\item (quasi-finite) fin. type + finite fibers
	\item (finite) $V = \Spec B$ $ \Rightarrow f ^{-1}(V) = \Spec A$, $A$ f.g. $B$-mod.
	\item (separated) $\Delta_{X/Y} : X \to X \times_Y X$ closed imm.
	\item (proper) separated + fin. type + univ. closed
\end{itemize}
\end{definition}

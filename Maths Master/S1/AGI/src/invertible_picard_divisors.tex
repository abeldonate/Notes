$f : X \to Y$, $\mathcal{L} $ invertible.

\begin{definition}
\begin{itemize}[topsep=-6pt, itemsep=0pt]
    \item (quasi-projective) $\exists$ immersion $ \iota : X \to \mathbb{P}^n_Y$ over $Y$.
    \item (projective) quasi-projective + proper.
    \item (very ample) $\mathcal{L} \cong \iota^* \mathcal{O}_{\mathbb{P}^n_Y}(1)$, $\iota : X \to \mathbb{P}^n_Y$ closed immersion over $Y$.
    \item (ample) $\ \forall  \mathcal{F} \in \Coh(X) \exists n_0 > 0$ s.t. $\mathcal{F} \otimes \mathcal{L}^{ \otimes n}  $ g.g. for $n \geq n_0$
\end{itemize}
\end{definition}

\begin{proposition}
very ample $ \Rightarrow $ ample
\end{proposition}

\begin{proposition}
\begin{itemize}[topsep=-6pt, itemsep=0pt]
    \item $\mathcal{L}, \mathcal{M}  $ ample $ \Rightarrow \mathcal{L} \otimes \mathcal{M} $ ample
    \item $\mathcal{L} $ ample, $\mathcal{M} $ g.g. $ \Rightarrow \mathcal{L} \otimes \mathcal{M}$ ample
    \item $\mathcal{L}$ ample $\mathcal{L}^{ \otimes n} \otimes \mathcal{M} \ \forall  n \geq  n_0$
    \item $\mathcal{L} $ very ample, $\mathcal{M} $ g.g. $ \Rightarrow $ $\mathcal{L} \otimes \mathcal{M}$ very ample
\end{itemize}
\end{proposition}

\begin{proposition} (X Noetherian) $\mathcal{L} $ ample
$ \Leftrightarrow  \mathcal{L}^{ \otimes n } $ ample $\ \forall  n > 0 $
$ \Leftrightarrow $ $\exists n > 0, \mathcal{L}^{ \otimes n } $ ample
\end{proposition}

\begin{proposition} $f : X \to \Spec A $ finite type, $A$ Noetherian. Then $\mathcal{L} $ ample $ \Leftrightarrow \exists n >0, \mathcal{L}^{ \otimes n} $ very ample.
\end{proposition}

\begin{proposition} $f : X \to \Spec A $ proper, $A$ Noetherian. Then $\mathcal{L} $ very ample $ \Rightarrow \mathcal{L} $ ample.
\end{proposition}

\begin{definition}$\Pic(X)$ group of iso. classes of inv.sheaves under $ \otimes $.
\end{definition}

\begin{definition}[Dimensions]
    \begin{itemize}[topsep=-6pt, itemsep=0pt]
        \item $\dim (X) \coloneqq \sup \{ n : Z_0 \subsetneq Z_1 \subsetneq \cdots  \subsetneq Z_n \subseteq X \} $ $Z_i$ irred. closed.
        \item $\codim (Z, X) \coloneqq \sup \{ n : Z = Z_0 \subsetneq \cdots  \subsetneq Z_n \subseteq X \} $ $Z_i$ irred. closed.
        \item $\height (\mathfrak{p}) \coloneqq \sup \{ n : \mathfrak{p}_0 \subsetneq \mathfrak{p}_1 \subsetneq \cdots  \subsetneq \mathfrak{p}_n = \mathfrak{p}\}$
    \end{itemize}
\end{definition}

\begin{proposition}
$\height (\mathfrak{p}) = \codim(V(\mathfrak{p}), \Spec A)$, $\dim (\Spec A) = \dim (A)$
\end{proposition}

\begin{proposition} $A$ integral f.t. over $k$. Then $\dim A = \text{tr.deg}_k (\text{Frac}(A)) = \dim (A/\mathfrak{p}) + \height (\mathfrak{p})$
\end{proposition}

\begin{proposition} [Krull's Hauptidealsatz] $A$ Noetherian, $a$ nonzerodivisor. Then $\mathfrak{p} \in V(a)$ with $\mathfrak{p}$ minimal $ \Rightarrow $ $\height (\mathfrak{p}) = 1$.
\end{proposition}

\begin{proposition} $A$ Noetherian, integral. Then $A = \bigcap_{\height (\mathfrak{p}) = 1} A_{\mathfrak{p}}$.
\end{proposition}

\begin{definition}[Regular] Noetherian + local + $\dim_{k} \mathfrak{m}/\mathfrak{m}^2 = \dim A$.
\end{definition}

\begin{proposition} ($A$ Noetherian local dim 1)
Then, $A$ regular $ \Leftrightarrow $ $A$ normal $ \Leftrightarrow $ $A$ dvr
\end{proposition}

\begin{definition}
\begin{itemize}[topsep=-6pt, itemsep=0pt]
    \item $\div : \{(\mathcal{L}, s )\} / \sim \to \Div(X), (\mathcal{L}, s) \mapsto \sum \nu _Z(s) Z$
    \item (effective) $n_Z \geq 0 \ \forall  Z$
    \item (principal) $D = \div (f)$ for some $f \in K(X)^*$
    \item (cartier) $D$ closed subscheme whose ideal sheaf is invertible
\end{itemize}
\end{definition}

\begin{definition}[R1] Regular in codim 1 $ \Leftrightarrow $ $\mathcal{O}_{X, x}$ regular $\forall x$ with $\dim (\mathcal{O}_{X,x}) = 1$.
\end{definition}

\begin{proposition} Normal + Noetherian $ \Rightarrow $ (R1)
\end{proposition}

\begin{proposition} $\Pic (X) \to \Cl(X), [\mathcal{L} ] \mapsto [\div (s)]$
\end{proposition}

\begin{proposition} $X$ integral Noetherian (R1) $ \Rightarrow \Pic(X) \hookrightarrow \Cl(X)$ injective.
\end{proposition}

\begin{definition}[Sheaf associated] $\mathcal{O}_X (D) : U \mapsto \{f \in k(X)^{\times } : \div \mid _U f + D\mid _U \geq 0\} \cup \{0\}$
\end{definition}

\begin{proposition} Integral + normal + Noetherian $ \Rightarrow \mathcal{O}_X(D) \in \QCoh(X)$.
\end{proposition}

\begin{proposition}
$\mathcal{O}_X (\div (\mathcal{L}, s)) \cong \mathcal{L}$, $1 \mapsto s$. $D$ locally principal $ \Rightarrow \mathcal{O}_X(D) $ invertible.
\end{proposition}

\begin{proposition} $ \mathcal{O}_X(D) $ invertible $ \Rightarrow D = \div ( \mathcal{O}_X(D), 1) $
\end{proposition}

\begin{proposition} 
   \begin{itemize}[topsep=-6pt, itemsep=0pt]
    \item $A$ Noetherian + integral + ($\height (\mathfrak{p}) = 1 \Rightarrow \mathfrak{p}$ princ.) $ \Rightarrow A $ factorial.
    \item $A$ factorial $ \Rightarrow A$ normal.
    \item $A$ regular + Noetherian + local $ \Rightarrow A $ factorial.
   \end{itemize}    
\end{proposition}

\begin{definition} $A$ local integral, $k = \text{frac}(A)$.
\begin{itemize}[topsep=-6pt, itemsep=0pt]
    \item $B \subseteq k$ local integral dominates $A \Leftrightarrow \mathfrak{m}_B \cap A = \mathfrak{m}_A$
    \item (vr) maximal with respect to domination.
    \item (dvr) vr + Noetherian + not field
\end{itemize}
\end{definition}

\begin{proposition}
$A$ dvr $ \Leftrightarrow A$ local integral + $\mathfrak{m}$ principal
\end{proposition}

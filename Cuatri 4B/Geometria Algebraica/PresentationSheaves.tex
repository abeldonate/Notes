\section[Intro]{Introduction}

\begin{frame}[fragile]{Definición de prehaz}
Sea $X$ un espacio topológico. Un prehaz de grupos abelianos consiste en:
\begin{enumerate}
  \item Una aplicación $U \mapsto \mathcal{F}(U)$
  \item Un mapa de restricción $\rho_{UV}: \mathcal{F}(U) \to \mathcal{F}(V)$ para $ V \subseteq U$ que preserve identidad y composición.
\end{enumerate} 

\incfig{presheaf}

Llamaremos \textit{secciones} a los elementos del grupo abeliano $s \in \mathcal{F}(U) $ 

\end{frame}


\begin{frame}[fragile]{Definición de haz}
Decimos, además, que el el prehaz es un haz si cumple las dos siguientes condiciones sobre un recubrimiento $\{U_i\}$ de $U$:
\begin{itemize}
  \item (Pegado) Si tfenemos secciones sobre $U_i$ que coinciden en las intersecciones, existe una sección global en $U$.
  \item (Localidad) Si tenemos dos secciones que coinciden en $U_i$, entonces son iguales.
\end{itemize}

\incfig{sheaf}

\end{frame}


\begin{frame}[fragile]{Ejemplo haz de funciones continuas}
Sea $X = \mathbb{R}$, consideramos el haz que envía cada abierto a las funciones continuas en el abierto $\mathcal{F}(U) = \mathcal{C}^0(U)$.
\incfig{sheafcont}
\end{frame}


\begin{frame}[fragile]{Ejemplo prehaz que no es un haz}

\begin{frame}[fragile]{Haz asociado}
Dado un prehaz $\mathcal{F}$, existe un único haz $\mathcal{F}'$ y un morfismo $\theta: \mathcal{F} \to \mathcal{F}'$  tal que para cada haz $\mathcal{G} $ y morphismo $\varphi $ hay un único $\psi $ tal que
\[\begin{tikzcd}
	{\mathcal{F}} & {\mathcal{F}'} \\
	& {\mathcal{G}}
	\arrow["\theta", from=1-1, to=1-2]
	\arrow["\varphi"', from=1-1, to=2-2]
	\arrow["\exists \psi", from=1-2, to=2-2]
\end{tikzcd}\]
conmuta.
\end{frame}

\begin{frame}[fragile]{Stalk}
El stalk de un haz (o prehaz) en un punto $P\in X$ se define como el límite directo de los entornos que contienen el punto:
\[
\mathcal{F}_P = \varinjlim_{U \ni P} \mathcal{F}(U) 
\]

\incfig{stalk}

\end{frame}


\begin{frame}[fragile]{Ejemplos de haces y stalks}

  Sea $X = \mathbb{R}$. Tenemos dos ejemplos de haces:
\begin{enumerate}
  \item $\mathcal{F}(U) = \{f: f\in \mathbb{R}[x]\}$. Entonces $\mathcal{F}_P \cong \mathbb{R}[x]$ 
  \item $\mathcal{F}(U) = \{f: f\in \mathcal{C}^{\omega}(U) \}$. Entonces $\mathcal{F}_P \cong \mathbb{R}[[x]]$ 
\end{enumerate}

\incfig{stalk}

\end{frame}
